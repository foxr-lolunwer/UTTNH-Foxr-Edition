add_namespace = FLTE_sp_scientist_event
country_event = { # specialization_land
	id = FLTE_sp_scientist_event.1
	title = FLTE_sp_scientist_event.1.title
	desc = FLTE_sp_scientist_event.1.desc
	is_triggered_only = yes
	immediate = {
		event_target:FLTE_scientist_event_target = {
			hidden_effect = {
				if = { limit = {
					has_scientist_level = {
						level > 1
						specialization = specialization_land } }
					add_scientist_level = {
						level = -1
						specialization = specialization_land }
					custom_effect_tooltip = FLTE_sp_scientist_event.1.tooltip
				}
				else = { log = FLTE_WARNING_101 }
			}

		}
	}
	option = { name = FLTE_sp_scientist_event.1.option_1 }
}
