FLTE_EFFECT_init_world_buildings = {
    hidden_effect = {
        if = {
            limit = { NOT = { has_global_flag = FLTE_GOLFLAG_init_world_buildings_done } }
            every_state = {
                if = {
                    limit = { check_variable = { building_level@infrastructure > 0 } }
                    set_temp_variable = { FLTE_state_building_add_num = building_level@infrastructure }
                    multiply_temp_variable = { FLTE_state_building_add_num = 3 }
                    add_building_construction = { type = infrastructure level = FLTE_state_building_add_num }
                }
                if = {
                    limit = { check_variable = { building_level@industrial_complex > 0 } }
                    set_temp_variable = { FLTE_state_building_add_num = building_level@industrial_complex }
                    multiply_temp_variable = { FLTE_state_building_add_num = 2 }
                    add_building_construction = { type = industrial_complex level = FLTE_state_building_add_num }
                }
                if = {
                    limit = { check_variable = { building_level@arms_factory > 0 } }
                    set_temp_variable = { FLTE_state_building_add_num = building_level@arms_factory }
                    # multiply_temp_variable = { FLTE_state_building_add_num = 1 }
                    add_building_construction = { type = arms_factory level = FLTE_state_building_add_num }
                }
                if = {
                    limit = { check_variable = { building_level@bunker > 0 } }
                    set_temp_variable = { FLTE_state_building_add_num = building_level@bunker }
                    # multiply_temp_variable = { FLTE_state_building_add_num = 1 }
                    add_building_construction = { type = bunker level = FLTE_state_building_add_num }
                }
                if = {
                    limit = { check_variable = { building_level@air_base > 0 } }
                    set_temp_variable = { FLTE_state_building_add_num = building_level@air_base }
                    # multiply_temp_variable = { FLTE_state_building_add_num = 1 }
                    add_building_construction = { type = air_base level = FLTE_state_building_add_num }
                }
                if = {
                    limit = { check_variable = { building_level@naval_base > 0 } }
                    set_temp_variable = { FLTE_state_building_add_num = building_level@naval_base }
                    # multiply_temp_variable = { FLTE_state_building_add_num = 1 }
                    add_building_construction = { type = naval_base level = FLTE_state_building_add_num }
                }
            }
            set_global_flag = FLTE_GOLFLAG_init_world_buildings_done
        }
    }
}
